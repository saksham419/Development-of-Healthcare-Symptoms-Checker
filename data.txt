  <script>
    // Theme toggle
    const switchEl = document.getElementById('themeSwitch');
    const body = document.body;
    function setTheme(theme){
      body.setAttribute('data-theme', theme);
      localStorage.setItem('theme', theme);
    }
    // initial theme
    const saved = localStorage.getItem('theme') || 'light';
    setTheme(saved);
    // set knob visually
    if(saved === 'dark') document.querySelector('.knob').style.left = '24px';

    switchEl.addEventListener('click', ()=> {
      const current = body.getAttribute('data-theme') === 'dark' ? 'light' : 'dark';
      setTheme(current);
      document.querySelector('.knob').style.left = current === 'dark' ? '24px' : '4px';
    });

    // Clear button
    document.getElementById('clearBtn').addEventListener('click', ()=> {
      const sel = document.getElementById('symptoms');
      for(let i=0;i<sel.options.length;i++){ sel.options[i].selected = false; }
    });
  </script>
